\subsection*{\href{https://www.overleaf.com/learn/latex/Footnotes}{Footnotes}}


\entry{25mm}{\textbackslash footnote\{\dots\}}{basic use}\\
\entry{25mm}{\textbackslash footnote[3]\{\dots\}}{micromanage foot num}\\
\entry{25mm}{\textbackslash thefootnote}{last foot \#}\\
\textit{Set style as follows:}\\
\code{\textbackslash renewc'md\{\textbackslash thefootnote\}\{\textbackslash roman\{footnote\}\}}\\

\textit{Use symbols rather than numbers (from \href{https://tex.stackexchange.com/questions/826/symbols-instead-of-numbers-as-footnote-markers}{here}):}\\
\href{http://mirrors.ctan.org/macros/latex/contrib/footmisc/footmisc.pdf}{\code{\textbackslash usepackage[symbol]\{footmisc\}}}\\
\code{\textbackslash renewc'md\{\textbackslash thefootnote\}\{\textbackslash fnsymbol\{footnote\}}\\
\entry{35mm}{\textbackslash footnote[<num>]\{text\}}{<num> $\in$: 1-9}\\
